# apps-pacman.txt
# Official Arch repository packages
# 
# Format: one package per line
# Lines starting with # are comments
# Everything after the first word is ignored (you can add comments inline)
#
# To add a package: just add a new line with the package name
# To disable a package: add # at the start of the line
#
# Example:
#   firefox          # web browser
#   # chromium       # disabled for now
